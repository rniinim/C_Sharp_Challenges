/* This solution for Russian roulette game */

bool alive = true;
Random rng = new Random();
int shot = 0;

while (alive)
{
    shot = rng.Next(1, 7);
    Console.WriteLine("Press enter to pull trigger");
    Console.ReadKey();

    if (shot == 6)
    {
        alive = false;
    }
    else
    {
        Console.WriteLine("Click!");
    }
}
Console.WriteLine("Boom! Game over");
