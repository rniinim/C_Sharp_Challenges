/* This is solution for Number password generator */

Random rng = new Random();

Console.Write("Give lenght of password: ");
int length = int.Parse(Console.ReadLine());
int temp;
string password = "";

for (int i = 0; i < length; i++)
{
    temp = rng.Next(0, 10);
    password += temp;
}
Console.WriteLine(password);
